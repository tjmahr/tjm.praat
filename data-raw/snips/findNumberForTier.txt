# Find the number of the (last) tier with a given name
procedure findNumberForTier: .target_tier$
  .tiers = Get number of tiers
  .result = 0

  for tier_i to .tiers
    tier_i_name$ = Get tier name: tier_i
      if tier_i_name$ == .target_tier$
        .result = tier_i
      endif
  endfor

endproc
